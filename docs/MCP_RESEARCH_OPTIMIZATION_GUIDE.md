const progressiveSearch = async (baseQuery: string) => {

    `${baseQuery}`,

    `${baseQuery} React Next.js`,
    `${baseQuery} React Next.js Tailwind CSS`,
    `${baseQuery} React Next.js Tailwind CSS TypeScript`

    }
  }
